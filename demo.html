      // Find recent sign-in failures
let threshold = 10;
SigninLogs
| where TimeGenerated > ago(1d)
| where ResultType != "0"
| summarize FailureCount = count() by UserPrincipalName, IPAddress
| where FailureCount > threshold
| order by FailureCount desc
| take 25
